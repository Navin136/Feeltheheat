CREATE DATABASE IF NOT EXISTS MASTER;
USE MASTER;
CREATE TABLE IF NOT EXISTS part_details(
    part_number INT PRIMARY KEY AUTO_INCREMENT,
    carbon DECIMAL(3,2),
    silicon DECIMAL(3,2),
    copper DECIMAL(3,2),
    tin DECIMAL(4,3),
    manganese DECIMAL(3,2),
    molybdenum DECIMAL(4,3),
    nickel DECIMAL(4,3),
    magnesium DECIMAL(4,3),
    mischmetal varchar(50),
    laddition varchar(50),
    inoculant varchar(30),
    flow_rate decimal(2,1),
    metalweight INT,
    temperature varchar(15),
    taddition varchar(50),
    pouring_time varchar(20),
    alloy varchar(20),
    treewt DECIMAL(4,2),
    grade varchar(50),
    alloyp DECIMAL(3,2),
    coveringp DECIMAL(3,2),
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE IF NOT EXISTS live(
    current_part INT,
    id INT AUTO_INCREMENT PRIMARY KEY,
    pattern_changed_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE IF NOT EXISTS holder(
    id INT AUTO_INCREMENT PRIMARY KEY,
    heat VARCHAR(50),
    carbon DECIMAL(3,2),
    silicon DECIMAL(3,2),
    copper DECIMAL(3,2),
    tin DECIMAL(4,3),
    manganese DECIMAL(3,2),
    molybdenum DECIMAL(4,3),
    nickel DECIMAL(4,3),
    titanium DECIMAL(4,3),
    last_updated DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE IF NOT EXISTS todayplan(
    id INT AUTO_INCREMENT PRIMARY KEY,
    todaydate DATE,
    plan_no INT(5) ,
    part INT(10),
    moulds INT(10),
    last_updated DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE IF NOT EXISTS chargemix(
    id INT AUTO_INCREMENT PRIMARY KEY,
    grade varchar(50),
    steelp INT(5),
    tinsteelp INT(5),
    greysteelp INT(5),
    boringsp INT(5),
    pigironp INT(5),
    returnsp INT(5),
    last_updated DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE IF NOT EXISTS composition(
    id INT AUTO_INCREMENT PRIMARY KEY,
    grade varchar(50),
    material varchar(20),
    carbon DECIMAL(3,2),
    silicon DECIMAL(3,2),
    copper DECIMAL(3,2),
    tin DECIMAL(4,3),
    manganese DECIMAL(3,2),
    molybdenum DECIMAL(3,2),
    nickel DECIMAL(3,2),
    titanium DECIMAL(3,2),
    last_updated DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE IF NOT EXISTS chargemixdisplay(
    id INT AUTO_INCREMENT PRIMARY KEY,
    runningpart varchar(10),
    last_updated DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
