USE MASTER;
CREATE TABLE part_details(
    part_number INT PRIMARY KEY AUTO_INCREMENT,
    carbon DECIMAL(3,2),
    silicon DECIMAL(3,2),
    copper DECIMAL(3,2),
    tin DECIMAL(4,3),
    manganese DECIMAL(3,2),
    molybdenum DECIMAL(4,3),
    nickel DECIMAL(4,3),
    magnesium DECIMAL(4,3),
    mischmetal varchar(50),
    laddition varchar(50),
    inoculant varchar(30),
    flow_rate decimal(2,1),
    metalweight INT,
    temperature varchar(15),
    taddition varchar(50),
    pouring_time varchar(20),
    alloy varchar(20),
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
CREATE TABLE live(
    current_part INT,
    id INT
);
CREATE TABLE holder(
    id INT PRIMARY KEY,
    heat VARCHAR(10),
    carbon DECIMAL(3,2),
    silicon DECIMAL(3,2),
    copper DECIMAL(3,2),
    tin DECIMAL(4,3),
    manganese DECIMAL(3,2),
    molybdenum DECIMAL(4,3),
    nickel DECIMAL(4,3),
    titanium DECIMAL(4,3),
    last_updated DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
);
